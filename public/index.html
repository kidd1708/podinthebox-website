// ========== API FUNCTIONS WITH CACHE BUSTING ========== async function loadEpisodes() { try { const timestamp = new Date().getTime(); const response = await fetch(`/api/episodes?t=${timestamp}`, { cache: 'no-store', headers: { 'Cache-Control': 'no-cache', 'Pragma': 'no-cache' } }); if (response.ok) { episodes = await response.json(); renderPublicSite(); if (document.getElementById('adminEpisodesList')) renderAdminEpisodes(); } else { console.error('Failed to load episodes'); } } catch (error) { console.error('Error loading episodes:', error); } } async function loadBlogPosts() { try { const timestamp = new Date().getTime(); const response = await fetch(`/api/blog?t=${timestamp}`, { cache: 'no-store', headers: { 'Cache-Control': 'no-cache', 'Pragma': 'no-cache' } }); if (response.ok) { blogPosts = await response.json(); console.log(`Loaded ${blogPosts.length} blog posts`); renderPublicSite(); if (document.getElementById('adminBlogList') && document.getElementById('adminDashboardArea').style.display === 'block') { renderAdminBlog(); } } else { console.error('Failed to load blog posts'); } } catch (error) { console.error('Error loading blog posts:', error); } } async function loadStories() { if (!adminToken) return; try { const timestamp = new Date().getTime(); const response = await fetch(`/api/stories?t=${timestamp}`, { cache: 'no-store', headers: { 'Cache-Control': 'no-cache', 'Pragma': 'no-cache', 'X-Admin-Token': adminToken } }); if (response.ok) { storySubmissions = await response.json(); if (document.getElementById('adminStoriesList')) renderAdminStories(); } else if (response.status === 401) { console.log('Session expired, please login again'); logoutAndClose(); } else { console.error('Failed to load stories'); } } catch (error) { console.error('Error loading stories:', error); } } async function forceRefreshAllData() { console.log("Force refreshing all data..."); blogPosts = []; episodes = []; storySubmissions = []; await loadEpisodes(); await loadBlogPosts(); if (adminToken && document.getElementById('adminStoriesList')) { await loadStories(); } console.log("Data refresh complete"); } async function loadAllData() { await loadEpisodes(); await loadBlogPosts(); } // Helper function for admin requests with cache busting async function adminFetch(url, options = {}) { const timestamp = new Date().getTime(); const separator = url.includes('?') ? '&' : '?'; const urlWithTimestamp = `${url}${separator}t=${timestamp}`; const headers = { 'Content-Type': 'application/json', 'Cache-Control': 'no-cache', 'Pragma': 'no-cache', ...options.headers }; if (adminToken) { headers['X-Admin-Token'] = adminToken; } const response = await fetch(urlWithTimestamp, { ...options, cache: 'no-store', headers }); if (response.status === 401) { showToast("Session expired. Please login again.", true); logoutAndClose(); } return response; }
